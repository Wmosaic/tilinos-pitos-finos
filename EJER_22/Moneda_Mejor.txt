2.2. CapturaFicheros()
	cuenta = 0
	Articulo obj_a = Nuevo Moneda()
	Moneda lista[50]
	archivo = Abrir(_...Direccion.txt)
	monedas = archivo.leer()
	
	para i = 0; i < monedas; i++ empezar
		lista[Cuenta] = monedas[i]

		cuenta++

3.1 confiDivisas()
   limS = 0

   Para i = 0 hasta iterador i en 1 a 1
      Setdivisas[i] = arreglo[i]
   Terminar

   tem_Div = Setdivisas.Quitar_ElementosRepetidos()

   Para i = 0 hasta tem_Div.len() i en 1 en 1
      Setdivisas[limS] = tem_Div[i]
      limS++
   Terminar
   regresar Setdivisas[limS]

3.2 calcu()
   Setdivisas[] = confiDivisas()
   Para j = 0; j < limS; j++ empezar
      dineros = 0
      Para i = 0; i < iterador; i ++
         Si Setdivisas[j] == arreglo[i].MetodoObtenerDivisa()
            dinero += guarda_can[i] * arreglo[i].MetodoObtenerValor()
      cuenta_val[j] = dinero
   Terminar
